# 新建一个scrapy项目
scrapy startproject myfirstscrapy

# 编写爬虫
cd myfirstscrapy
scrapy genspider dmoz_spider dmoz.org
vi myfirstscrapy/spider/dmoz_spider

# 运行爬虫
scrapy crawl dmoz_spider --nolog

scrapy list
